import { getData } from '@libs/data'
import { getConfig } from '@libs/config'
import { allTokens, allCustomPropsSassVariables } from '@libs/sass-variables'
import { getVersionedDocsPath } from '@libs/path'

export const customPropFigure = (customProp) => (
  <figure class="mb-none" aria-label={customProp}>
    <button class="border-none p-none color-copy w-100 ratio-1x1 bg-body" data-clipboard-text={customProp} data-bs-toggle="tooltip" data-bs-title="Copy to clipboard">
      <svg viewBox="0 0 100 100" class="rounded" aria-hidden="true" preserveAspectRatio="xMidYMid meet">
        <rect fill={customProp} width="100" height="100"/>
      </svg>
      <span class="visually-hidden">Copy variable name {customProp}</span>
    </button>
    <figcaption class="py-2xsmall">
      <var class="user-select-all">{customProp}</var>
    </figcaption>
  </figure>
)

export const getSurfaceColorFor = (colorName) => {
  if(colorName.includes('-action-')) {
    return `var(--bs${colorName.replace('$ouds-color-content-on-', '-color-')})`
  }
  return `var(--bs${colorName.replace('$ouds-color-content-on-', '-color-surface-')})`
}
