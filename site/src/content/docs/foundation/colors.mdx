import { extractRootName, mergeTokensByMode } from '@components/shortcodes/ColorTable.astro'

export const getSurfaceColorFor = (colorName) => {
  if(colorName.includes('-action-')) {
    return `var(--bs${colorName.replace('$ouds-color-content-on-', '-color-')})`
  }
  return `var(--bs${colorName.replace('$ouds-color-content-on-', '-color-surface-')})`
}

This page lists all the colors available in OUDS Web and their usage, organized by their category. All colors are presented in their light and dark mode for reference, but they must be used through their custom property or implementing utilities that will dynamically conform to the [color mode]([[docsref:/foundation/color-modes]]).

## Background

These colors are used for whole pages and structurally meaningful sections backgrounds.

These colors can be used with [our background utilities]([[docsref:/utilities/background]]).

<Callout type="info" title="Usage reference">
  Please refer to the [background colors design reference]() for more information on usage.
</Callout>

<ColorTable tokenFilter={(token) => /\$ouds-color-bg-.*-(light|dark)/.test(token.name)} />

## Surface

These colors should be used as background for elements that are on another background.

These colors can be used with [our background utilities]([[docsref:/utilities/background]]). Some of these surface colors have an opacity value and are designed to come on top of another color, we explain [how to achieve that here]([[docsref:/utilities/background#multiple-backgrounds]]).

<ColorTable tokenFilter={(token) => /\$ouds-color-surface-((?!status)).*-(light|dark)/.test(token.name)} />

### Status

These surface colors are used specifically to indicate a status.

Like other surface colors, they can be used with [our background utilities]([[docsref:/utilities/background]]).

<ColorTable tokenFilter={(token) => /\$ouds-color-surface-status.*-(light|dark)/.test(token.name)} />

## Content

These colors are used for informative but non-interactive content such as texts and icons.

These colors can be used with [our text color utilities]([[docsref:/utilities/colors]]).

<ColorTable tokenFilter={(token) => /\$ouds-color-content((?!-on-)).*-(light|dark)/.test(token.name)} />

### Content on other colors

These content colors are used to guarantee maximum legibility for contents displayed on a background or surface color.

<Callout type="info">
  In this table we display an icon with the content color on their target background color.
</Callout>
